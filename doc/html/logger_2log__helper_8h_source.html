127 template<typename T>
128 void write_impl(const T& val) {
129 constexpr bool is_suitable = std::is_integral_v<T> || std::is_floating_point_v<T>;
130 if constexpr (is_suitable) {
131 std::array<char, 100> buffer{};
132 auto&& [ptr, ec] = std::to_chars(buffer.data(), buffer.data() + buffer.size(), val);
133 write_impl(buffer.data(), std::size_t(ptr) - std::size_t(buffer.data()));
134 }
135 static_assert(is_suitable, "<--- hope::logger: An attempt was made to write value of unregistered type.");
136 }
