34 template <typename T>
35 constexpr void swap(T&& l, T&& r) {
36 auto temp{ r };
37 r = static_cast<T&&>(l);
38 l = static_cast<T&&>(temp);
39 }

41 template<std::size_t... Is, typename... Ts>
42 constexpr auto sort_impl(std::index_sequence<Is...> sequence, type_list<Ts...> list) {
43 constexpr auto listSize = size(list);
44 std::array<sort_helper, listSize> sizes{ sort_helper(Is, sizeof(Ts))... };
45
46 // should be replaced with std::sort in cxx20
47 for (size_t i{ 0 }; i < listSize; ++i) {
48 auto it = std::max_element(std::begin(sizes) + i, std::end(sizes));
49 size_t pos = std::distance(std::begin(sizes), it);
50 if (i != pos)
51 swap(sizes[i], sizes[pos]);
52 }
53 return sizes;
54 }
