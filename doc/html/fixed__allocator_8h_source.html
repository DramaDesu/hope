120 inline void* fixed_allocator::allocate() noexcept {
121 if (m_last_allocated->free_blocks_count == 0
122 && !update_alloc_chunk()) {
123 try {
124 m_chunk_list.push_back(create_new_chunk());
125 } catch (const std::exception&) {
126 assert(false);
127 return nullptr;
128 }
129 m_last_allocated = &m_chunk_list.back();
130 // If std::vector were reallocated m_last_deallocated could be invalid
131 // i decide to assign this var to the middle of vec is not bad idea
132 m_last_deallocated = &m_chunk_list[m_chunk_list.size() / 2];
133 }
134 return m_last_allocated->allocate(m_block_size);
135 }

137 inline void fixed_allocator::deallocate(void* ptr) noexcept {
138 if (m_last_deallocated == nullptr
139 || !can_be_deallocated(*m_last_deallocated, ptr))
140 update_dealloc_chunk(ptr);
141 m_last_deallocated->deallocate(ptr, m_block_size);
142 update_free_chunk();

149 inline void fixed_allocator::update_dealloc_chunk(void* ptr) noexcept {
150 const auto chunk_it = std::find_if(std::begin(m_chunk_list), std::end(m_chunk_list),
151 [=](const auto& ch) {
152 return can_be_deallocated(ch, ptr);
153 });
154 assert(chunk_it != std::end(m_chunk_list));
155 m_last_deallocated = &*chunk_it;
156 }

158 inline bool fixed_allocator::update_alloc_chunk() noexcept {
159 const auto chunk_it = std::find_if(std::begin(m_chunk_list), std::end(m_chunk_list),
160 [=](const auto& ch) {
161 return ch.free_blocks_count > 0;
162 });
163 const bool isFound = chunk_it != std::end(m_chunk_list);
164 if (isFound)
165 {
166 m_last_allocated = &*chunk_it;
167 if (m_free_block == m_last_allocated)
168 m_free_block = nullptr;
169 }
170 return isFound;
171 }

173 inline bool fixed_allocator::can_be_deallocated(const chunk& chunk, void* ptr) const noexcept {
174 return ptr >= chunk.data && ptr < chunk.data + m_block_size * m_blocks_in_chunk;

183 inline void fixed_allocator::update_free_chunk() noexcept {
184 if (m_last_deallocated->free_blocks_count != m_blocks_in_chunk)
185 return;
186 if(m_free_block != nullptr && m_chunk_list.size() > 1){
187 const std::size_t dist = m_last_deallocated - m_chunk_list.data();
188 const auto begin = std::begin(m_chunk_list) + dist;
189 delete[] begin->data;
190 m_chunk_list.erase(begin);
191 m_last_allocated = m_last_deallocated = &m_chunk_list[m_chunk_list.size() / 2];
192 } else if (m_free_block == nullptr){
193 m_free_block = m_last_deallocated;
194 }
195 }

197 inline void fixed_allocator::steal_resources(fixed_allocator&& rhs) noexcept {
198 m_block_size = rhs.m_block_size;
199 m_blocks_in_chunk = rhs.m_blocks_in_chunk;
200 m_chunk_list = std::move(rhs.m_chunk_list);
201 assert(rhs.m_chunk_list.empty());
202 m_last_allocated = rhs.m_last_allocated;
203 m_last_deallocated = rhs.m_last_deallocated;
204 m_free_block = rhs.m_free_block;
205 rhs.m_free_block = rhs.m_last_allocated = rhs.m_last_deallocated = nullptr;
206 }
