29 template <typename T>

34 bool add(BaseType* link) noexcept {
35 const auto index = find_index(link);
36 if (index < holder_size) {
37 links[index] = link;
38 return true;
39 }
40 return false;

43 bool remove(BaseType* link) noexcept {
44 for (auto& link_ref : links) { // NOTE! we should not try to cast links one more time; it is assumed that the array contains only properly casted entities
45 if (link == link_ref) {
46 link_ref = nullptr;
47 return true;
48 }
49 }
50 return false;
51 }

67 template <typename T, typename NativeT = std::decay_t<T>>
68 [[nodiscard]] constexpr T* get_impl() const noexcept {
69 static_assert(contains<NativeT>(types));
70 constexpr std::size_t Index = find<NativeT>(types);
71 if constexpr (std::is_convertible_v<BaseType*, NativeT*>)
72 return static_cast<T*>(links[Index]); // in case if NativeT inherit BaseType via virtual (just virtual inheritance)
73 else // Base cannot be unambiguous converted to the derived
74 return dynamic_cast<T*>(links[Index]);
75 }
76
77 std::size_t find_index(BaseType* link) noexcept {
78 // ReSharper disable once CppEntityUsedOnlyInUnevaluatedContext
79 return find_if(types, [&](auto&& holder) {
80 using type = typename std::decay_t<decltype(holder)>::Type;
81 constexpr std::size_t index = find<type>(types);
82 return links[index] == nullptr && dynamic_cast<type*>(link) != nullptr;
83 });
84 }
85
86 constexpr static type_list<Links...> types{ };
87 constexpr static std::size_t holder_size{ size(types) };

89 link_list links{ };
90 };
