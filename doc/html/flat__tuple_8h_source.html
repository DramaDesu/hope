242 template<typename T, std::size_t I>
243 [[nodiscard]] constexpr auto deduce_type() const noexcept {
244 // we cannot unambiguous determine holding type, thus try to cast to value and ref, and const ref...
245 if constexpr (std::is_base_of_v<indexed_value<T, I>, self_t>)
246 return type_holder<T>{};
247 else if constexpr (std::is_base_of_v<indexed_value<T &, I>, self_t>)
248 return type_holder<T &>{};
