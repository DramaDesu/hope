203 template<typename T, std::size_t I>
204 [[nodiscard]] constexpr auto deduce_type() const noexcept {
205 // we cannot unambiguous determine holding type, thus try to cast to value and ref, and const ref...
206 if constexpr (std::is_base_of_v<indexed_value<T, I>, self_t>)
207 return type_holder<T>{};
208 else if constexpr (std::is_base_of_v<indexed_value<T&, I>, self_t>)
209 return type_holder<T&>{};
210 else
211 return type_holder<const T&>{};
212 }
