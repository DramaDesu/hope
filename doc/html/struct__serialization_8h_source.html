23 template <typename T, typename... CustomSerializer>
24 class pod_serializer final {
25 constexpr static std::size_t fields_count{ compute_field_count_recursive_constexpr<T>() };
26 using bit_mask = std::bitset <fields_count>;
27 using serializer_holder_t = custom_serializer_holder<CustomSerializer...>;
28 using serializer_map = std::unordered_map<std::type_index, std::any>;
29 public:
30 pod_serializer(T& serializable_struct, double _fp_accuracy = std::numeric_limits<double>::epsilon())
31 : m_value(serializable_struct)
32 , m_fp_accuracy(_fp_accuracy) {}

34 void serialize(package& pack) {
35 const auto mask_begin_pos = pack.reserve(get_bit_mask_actual_size());
36 std::size_t index{ 0 };
37 serialize(m_value, m_prev_value, pack, index);
38 const auto struct_end_pos = pack.seek(mask_begin_pos);
39 write_mask( pack);
40 pack.seek(struct_end_pos);
41 m_first_pack = false;
42 m_prev_value = m_value;
43 }
44
45 void deserialize(package& pack) {
46 read_mask(pack);
47 std::size_t index{ 0 };
48 deserialize(m_value, pack, index);
49 }

51 private:
52 template<typename TValue>
53 void serialize(const TValue& value, const TValue& prev_value, package& pack, std::size_t& cur_index) {
54 if constexpr (serializer_holder_t::template has<TValue>())
55 serialize_custom_type(value, prev_value, pack, cur_index);
56 else
57 serialize_impl(value, prev_value, pack, cur_index);
58 }

60 template <typename TValue>
61 void serialize_custom_type(const TValue& value, const TValue& prev_value, package& pack, std::size_t& cur_index) {
62 using serializer_t = typename decltype(serializer_holder_t::template serializer<TValue>())::Type;
63 auto&& serializer_any = extract<TValue, serializer_t>(m_serializer_map);
64 auto&& serializer = std::any_cast<serializer_t>(serializer_any);
65 m_mask[cur_index] = serializer(value, prev_value, pack);
66 ++cur_index;
67 }

69 template<typename TValue>
70 std::enable_if_t<std::is_class_v<TValue>>
71 serialize_impl(const TValue& value, const TValue& prev_value, package& pack, std::size_t& cur_index) {
72 auto&& tuple = tuple_from_struct(value, field_policy::reference{});
73 auto&& prev_tuple = tuple_from_struct(prev_value, field_policy::reference{});
74
75 for_each(tuple, prev_tuple, [&](auto&& field, auto&& prev_field) {
76 serialize(field, prev_field, pack, cur_index);
77 });
78 }

80 template<typename TValue>
81 std::enable_if_t<!std::is_class_v<TValue>>
82 serialize_impl(const TValue& value, const TValue& prev_value, package& pack, std::size_t& cur_index) {
83 const bool changed = m_first_pack || !is_equal(value, prev_value);
84 if (changed)
85 pack.write(value);
86 m_mask[cur_index] = changed;
87 ++cur_index;
88 }

92 if constexpr (serializer_holder_t::template has<TValue>())
93 deserialize_custom_type(value, pack, cur_index);
94 else
95 deserialize_impl(value, pack, cur_index);

98 template <typename TValue>
99 void deserialize_custom_type(TValue& value, package& pack, std::size_t& cur_index) {
100 if (m_mask[cur_index]) {
101 using deserializer_t = typename decltype(serializer_holder_t::template deserializer<TValue>())::Type;
102 auto&& deserializer_any = extract<TValue, deserializer_t>(m_deserializer_map);
103 auto&& deserializer = std::any_cast<deserializer_t>(deserializer_any);
104 deserializer(value, pack);
105 }
106 ++cur_index;
107 }

109 template<typename TValue>
110 std::enable_if_t<std::is_class_v<TValue>>
111 deserialize_impl(TValue& value, package& pack, std::size_t& cur_index) {
112 auto&& tuple = tuple_from_struct(value, field_policy::reference{});
113
114 for_each(tuple, [&](auto&& field) {
115 deserialize(field, pack, cur_index);
116 });
117 }

119 template<typename TValue>
120 std::enable_if_t<!std::is_class_v<TValue>>
121 deserialize_impl(TValue& value, package& pack, std::size_t& cur_index) {
122 if (m_mask[cur_index])
123 value = pack.read<TValue>();
124 ++cur_index;
125 }

127 template <typename V>
128 [[nodiscard]] bool is_equal(const V& left, const V& right) const {
129 if constexpr (std::is_floating_point_v<V>)
130 return std::abs(left - right) < m_fp_accuracy;
131 return left == right;
132 }

134 void write_mask(package& pack) {
135 if constexpr (fields_count <= std::numeric_limits<unsigned long>::digits)
136 pack.write(m_mask.to_ulong());
137 else
138 pack.write(m_mask.to_ullong());
139 }

141 template<typename MaskType>
142 void read_mask_impl(package& pack) {
143 auto&& mask = pack.read<MaskType>();
144 m_mask = bit_mask(mask);
145 }
146
147 void read_mask(package& pack) {
148 if constexpr (fields_count <= std::numeric_limits<unsigned long>::digits)
149 read_mask_impl<unsigned long>(pack);
150 else
151 read_mask_impl<unsigned long long>(pack);
152 }
153
154 [[nodiscard]] static constexpr auto get_bit_mask_actual_size() {
155 if constexpr (fields_count <= std::numeric_limits<unsigned long>::digits)
156 return sizeof (unsigned long);
157 return sizeof(unsigned long long);
158 }

160 template<typename TValue, typename TSerializer>
161 [[nodiscard]] std::any& extract(serializer_map& map) const {
162 auto&& type_index = std::type_index(typeid(TValue));
163 auto&& it = map.find(type_index);
164 if (it == std::end(map))
165 it = map.emplace(type_index, TSerializer{}).first;
166 return it->second;
167 }

169 T& m_value;
170 T m_prev_value;
171 bool m_first_pack{ true };
172 bit_mask m_mask;
173 serializer_map m_serializer_map;
174 serializer_map m_deserializer_map;
175 const double m_fp_accuracy;
176
177 static_assert(fields_count <= std::numeric_limits<unsigned long long>::digits, "Men, you can't use this serializer for structure with 65 fields and more.");
178 };
