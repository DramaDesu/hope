77 inline void chunk::init(std::size_t block_size, uint8_t blocks_count) noexcept {
78 data = new(std::nothrow) uint8_t[block_size * std::size_t(blocks_count)];
79 assert(data != nullptr);
80 free_blocks_count = blocks_count;
81 auto* begin = data;
82 for (uint8_t i{ 0 }; i < blocks_count; ++i, begin += block_size)
83 *begin = i + 1;

86 inline void* chunk::allocate(std::size_t block_size) noexcept {
87 assert(is_allocation_valid(block_size));
88 auto* const result = data + std::size_t(first_free_block) * block_size;
89 first_free_block = *result;
90 --free_blocks_count;
91 return result;

94 inline void chunk::deallocate(void* ptr, std::size_t block_size) noexcept {
95 assert(is_deallocation_valid(ptr, block_size));
96 const auto byte_index = std::size_t(static_cast<uint8_t*>(ptr) - data);
97 const auto block_index = uint8_t(byte_index / block_size);
98 data[byte_index] = first_free_block;
99 first_free_block = block_index;
100 ++free_blocks_count;
101 }

103 inline bool chunk::is_deallocation_valid(void* ptr, std::size_t block_size) const noexcept {
104 return data != nullptr
105 && ptr != nullptr
106 && free_blocks_count < std::numeric_limits<uint8_t>::max()
107 && block_size > 0
108 && (static_cast<uint8_t*>(ptr) - data) % block_size == 0;
109 }
